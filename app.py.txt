# 🔹 curs BNR
def get_bnr(date):
    try:
        url = f"https://api.exchangerate.host/{date}?base=EUR&symbols=RON"
        r = requests.get(url, timeout=10).json()
        return float(r["rates"]["RON"])
    except:
        return 5.08

# 🔹 reguli contabile
def map_account(text):
    t = text.lower()

    if "coolunity" in t:
        return "704"
    if "omv" in t:
        return "6022"
    if "leasing" in t:
        return "167"
    if "taxa" in t or "asigur" in t:
        return "635"
    if "delegare" in t:
        return "625"

    return "401"

# 🔹 parser PDF
def parse_pdf(file):
    data = []

    with pdfplumber.open(file) as pdf:
        text = "\n".join([p.extract_text() for p in pdf.pages if p.extract_text()])

    blocks = re.split(r'\n(?=\d{1,2}\s\w{3}\.\s\d{4})', text)

    for b in blocks:
        if "€" not in b:
            continue

        date_match = re.search(r'(\d{1,2}\s\w{3}\.\s\d{4})', b)
        if not date_match:
            continue

        try:
            date_str = date_match.group(1)
            date = datetime.strptime(date_str, "%d %b. %Y").strftime("%Y-%m-%d")
        except:
            continue

        eur = re.findall(r'€\s?([\d\.,]+)', b)
        ron = re.findall(r'([\d\.,]+)\sRON', b)

        eur = float(eur[0].replace(",", "")) if eur else 0
        ron = float(ron[0].replace(",", "")) if ron else 0

        if "De la" in b:
            tip = "inc"
        elif "Către" in b:
            tip = "pl"
        elif "RON" in b:
            tip = "fx"
        else:
            continue

        data.append((date, tip, eur, ron, b[:80]))

    return data

@app.route("/", methods=["GET", "POST"])
def home():
    if request.method == "POST":
        f = request.files.get("file")

        if not f:
            return "Nu ai incarcat fisier", 400

        tx = parse_pdf(f)

        out_file = "saga.txt"

        with open(out_file, "w", encoding="utf-8") as g:
            nr = 1

            for d, t, eur, ron, exp in tx:

                # INCASARE
                if t == "inc":
                    g.write(f"{d};OP;{nr};{exp};5124.3;704;{eur};EUR\n")
                    nr += 1

                # PLATA
                elif t == "pl":
                    cont = map_account(exp)
                    g.write(f"{d};OP;{nr};{exp};{cont};5124.3;{eur};EUR\n")
                    nr += 1

                # SCHIMB VALUTAR
                elif t == "fx":
                    kurs = get_bnr(d)
                    diff = round(eur * kurs - ron, 2)

                    g.write(f"{d};NC;{nr};Schimb EUR-RON;581.2;5124.3;{eur};EUR\n")
                    nr += 1

                    g.write(f"{d};NC;{nr};Schimb EUR-RON;5121.3;581.2;{ron};RON\n")
                    nr += 1

                    if diff > 0:
                        g.write(f"{d};NC;{nr};Pierdere curs;665;581.2;{diff};RON\n")
                    elif diff < 0:
                        g.write(f"{d};NC;{nr};Castig curs;581.2;765;{abs(diff)};RON\n")

                    nr += 1

        return send_file(out_file, as_attachment=True)

    return render_template_string(HTML)
